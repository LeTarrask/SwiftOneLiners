Fibonacci - Recursively
func fibonacci(at index: Int) -> Int {
    return index <= 2 ? 1 : fibonacci(at: index-1) + fibonacci(at: index-2)
}

Multiples of 3 and 5
func sumMultiplesOfThreeAndFive(under upperBound: Int) -> Int {
    return (0..<upperBound).lazy.filter { $0.isMultiple(of: 3) || $0.isMultiple(of: 5) }.reduce(0, +)
}

Euclidean Algorithm - Recursively
func gcd(_ a: Int, _ b: Int) -> Int {
    return b == 0 ? a : gcd(b, a % b)
}
Euclidean Algorithm - Iteratively

TREES
// Tree Structure

class TreeMember {
    init() {
        assert(self is Node || self is Leaf)
    }
}

class Node: TreeMember {
    var children: [TreeMember]
    
    init(children: [TreeMember]) {
        self.children = children
        super.init()
    }
}

class Leaf: TreeMember {
    var id: String
    
    init(id: String) {
        self.id = id
        super.init()
    }
}
Depth First Traversal
func traverseTree(root: TreeMember) -> [Leaf] {
    return root is Leaf ? [root as! Leaf] : (root as! Node).children.lazy.flatMap(traverseTree)
}

Strings
Palindromes
func isPalindrome(_ string: String) -> Bool {
    return string.lowercased() == String(string.lowercased().reversed())
}
Anagrams
func areAnagrams(_ firstWord: String, secondWord: String) -> Bool {
    return firstWord.lowercased().sorted() == secondWord.lowercased().sorted()
}
Duplicate Characters
func duplicateCharacters(_ string: String) -> Set<Character> {
    return Set(string.filter({ string.firstIndex(of: $0) != string.lastIndex(of: $0) }))
}
